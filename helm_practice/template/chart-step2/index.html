1. 함수와 파이프라인


1-1) Chart Template 생성

helm create mychart

templates 폴더 안에 불필요한 파일 삭제

rm -rf deployment.yaml hpa.yaml ingress.yaml serviceaccount.yaml service.yaml tests

1-2) values.yaml에 해당 속성 추가

func:
  enabled: true
  
pipe:
  log: info

1-3) configmap 추가

vi cm1.yaml

apiVersion: v1
kind: ConfigMap
metadata:
  name: function-and-pipeline
data:
  Function_Argument: 
  quote: {{ quote .Values.func.enabled }}      # quote(arg1)
  include: {{ include "mychart.fullname" . }}  # include(arg1, arg2)

  Function_Quote: 
  function_case1: {{ .Values.func.enabled }}  
  function_case2: "{{ .Values.func.enabled }}"
  function_case3: {{ quote .Values.func.enabled }}

  Pipeline: 
  quote: {{ .Values.pipe.log | quote}} 
  upper.quote: {{ .Values.pipe.log | upper | quote}}
  repeat.upper.quote: {{ .Values.pipe.log | repeat 2 | upper | quote}}



2. 제어 흐름

2-1) values.yaml에 해당 속성 추가

dev:
  env: dev
  log: info

qa:
  env: qa
  log: info
  
prod:
  env: prod
  log:

2-2) configmap 추가

If 문에서 False 인 조건들

  Number:0, String: "", List: [], Object: {}, Boolean: false, Null

If 문에 쓰이는 함수들

  and(&&), or(||), ne(!=), not(!), eq(=), ge(>=), gt(>), le(<=), lt(<), default, empty

vi cm2.yaml

apiVersion: v1
kind: ConfigMap
metadata:
  name: flow-if
data:
  dev:
    env: {{ .Values.dev.env }}
    {{ if eq .Values.dev.env "dev" }}
    log: debug
    {{ else if .Values.dev.env }}
    log: {{ .Values.dev.log }}
    {{ else }}
    log: error
    {{ end }}

  qa:
    env: {{ .Values.qa.env }}
{{ if eq .Values.qa.env "qa" }}
    log: debug
{{ else if .Values.qa.env }}
    log: {{ .Values.qa.log }}
{{ else }}
    log: error
{{ end }}
    
  prod:
    env: {{ .Values.prod.env }}
    {{- if eq .Values.prod.env "qa" }}
    log: debug
    {{- else if .Values.prod.env }}
    log: {{ .Values.prod.log }}
    {{- else }}
    log: error
    {{- end }}

vi cm-with.yaml

apiVersion: v1
kind: ConfigMap
metadata:
  name: flow-with
data:
  dev:
  {{- with .Values.dev }}
    env: {{ .env }}
    log: {{ .log }}
  {{- end }}

  qa:
  {{- with .Values.qa }}
    env: {{ .env }}
    log: {{ .log }}
  {{- end }}
  
  prod:
  {{- with .Values.prod }}
    env: {{ .env }}
    log: {{ .log }}
  {{- end }}

2-1) values.yaml에 해당 속성 추가

data:
  - a
  - b
  - c

vi cm-range.yaml

apiVersion: v1
kind: ConfigMap
metadata:
  name: flow-range
data:
  yaml:
  {{- .Values.data | toYaml | nindent 2 }}

  range:
  {{- range .Values.data }}
  - {{ . }}
  {{- end }}

  range-quote:
  {{- range .Values.data }}
  - {{ . | quote }}
  {{- end }}

  range-upper-quote:
  {{- range .Values.data }}
  - {{ . | upper | quote }}
  {{- end }}



3. 여러가지 함수들

3-1) values.yaml에 해당 속성 추가

default:
  nil: 
  list: []
  object: {}
  number: 0
  string: ""
  boolean: false
  
ternary: 
  case1: true
  case2: false
  
print:
  data1: "a"
  data2: "b"

3-2) configmap 추가

vi cm3.yaml

apiVersion: v1
kind: ConfigMap
metadata:
  name: string-func
data:
  default: # Number:0, String: "", List: [], Object: {}, Boolean: false, Null
    nil: {{ .Values.default.nil | default "default" }}
    list: {{ .Values.default.list | default (list "default1" "default2") | toYaml | nindent 6}}
    object: {{ .Values.default.object | default "default:1" | toYaml | nindent 6 }}
    number: {{ .Values.default.number | default "default" }}
    string: {{ .Values.default.string | default "default" }}
    boolean: {{ .Values.default.boolean | default "default" }}
    
  ternary:
    case1: {{ .Values.ternary.case1 | ternary "ture" "false" }}
    case2: {{ .Values.ternary.case2 | ternary "ture" "false" }}

  print:
    print: {{ print "Hard Cording" }}
    printf: {{ printf "%s-%s" .Chart.Name .Chart.Version }}
    
  trim:
    trim: {{ trim "  hello " }}
    trimSuffix: {{ trimSuffix "-" "hello-" }}

  random:
    randAlphaNum: {{ randAlphaNum 5 }}   # 0-9a-zA-Z
    randAlpha: {{ randAlpha 5 }}         # a-zA-Z
    randNumeric: {{ randNumeric 5 }}     # 0-9
    randAscii: {{ randAscii 5 }}         # ASCII characters

  indent:
    indent: {{- .Values.data | toYaml | indent 4 }}
    nindent: {{- .Values.data | toYaml | nindent 4 }}
    
  trunc: {{ trunc 5 "hello world" }}
  replace: {{ "hello world" | replace " " "-" }}
  contains: {{ contains "cat" "catch" }}
  b64enc: {{ b64enc "hello" }}
